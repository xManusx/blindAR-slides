\def\C++{%
\ifmonospace%
    C++%
\else%
    C\kern-.1em\raise.30ex\hbox{\smaller{++}}%
\fi%
\spacefactor1000 }

\def\Csharp{%
\ifmonospace%
    \Csharp%
\else%
    C\kern-.1em\raise.30ex\hbox{\smaller{\#}}%
\fi%

			\begin{flushright}
			\LARGE#1
			\\
			\normalsize#2
			\end{flushright}
		\end{minipage}
	\end{frame}
}
